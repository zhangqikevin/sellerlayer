response = requests.post(
    "https://api.sellerlayer.com/openapi/v2/markets/search",
    headers={"Authorization": "Bearer YOUR_KEY"},
    json={"categoryPath": ["Electronics", "Headphones"], "dateRange": "30d"}
)

market = response.json()
print(f"Monthly demand: {market['totalDemand']}")
print(f"Avg gross margin: {market['avgGrossMargin']}%")
